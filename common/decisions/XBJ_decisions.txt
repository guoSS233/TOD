		allowed = { tag = XBJ }
		visible = { always = yes }
		available = {
			has_political_power > 50
		}
		cost = 50
		complete_effect = {
			set_temp_variable = { change_XBJ_officers_influence_by = 5 }

			set_temp_variable = { change_XBJ_landlords_support_by = 5 }
			XBJ_officers_update = yes
			XBJ_landlords_update = yes
		}
	}
}
